        Math.abs(-10.1); // returns 10.1 absValue of -10.1 is 10.1
        Math.pow(4, 2); // returns 16 because 4^2 = 16
        
        Math.floor(4.7); // returns 4 because 4.7 rounds down to 4
        Math.ceil(4.3); // returns 5 because 4.3 rounds up to 5
        
        Math.sin(0); // returns 0 because sin of 0 radians is 0
        Math.cos(0); // returns 1 because cos of 0 radians is 1
        Math.tan(0); // returns 0 because tan of 0 radians is 0
        Math.toDegrees(3.14); // returns ~180 because pi radians is 180 degrees
        
        Math.min(3, 5.2); // returns 3 because 3 is less than 5.2
        Math.max(3, 5.2); // returns 5.2 because 5.2 is greater than 
    }
}

Solution 1

import java.util.Scanner;
public class CrazyMath {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        
        System.out.println("Enter angle 1:");
        double angle1 = input.nextDouble();
        
        System.out.println("Enter angle 2:");
        double angle2 = input.nextDouble();
        
        double higherAngle = Math.max(angle1, angle2);
        double inRadians = Math.toRadians(higherAngle);
        double cos = Math.cos(inRadians);
        double absVal = Math.abs(cos);
        
        System.out.println("Answer: " + absVal);
    }
}

public static double avg(double... numbers) {
    double sum = 0;
    for (double x: numbers) {
        sum += x;
    }
    return sum / numbers.length;
}
